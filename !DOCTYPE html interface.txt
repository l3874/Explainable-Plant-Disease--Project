      html += `<span><i class="fas fa-microchip"></i> TensorFlow</span>`;
      modelInfoBadge.innerHTML = html;
    }
    modelSelect.addEventListener('change', updateModelInfo);
    updateModelInfo();

    // ---------- preview ----------
    function updatePreview(file) {
      if (!file) {
        previewImage.classList.add('hidden');
        placeholderPreview.classList.remove('hidden');
        diagnoseBtn.disabled = true;
        currentImageFile = null;
        currentImageDataURL = null;
        return;
      }
      const reader = new FileReader();
      reader.onload = (e) => {
        currentImageDataURL = e.target.result;
        previewImage.src = currentImageDataURL;
        previewImage.classList.remove('hidden');
        placeholderPreview.classList.add('hidden');
        diagnoseBtn.disabled = false;
        currentImageFile = file;
        resetResults();
        // show original in side-by-side
        originalImage.src = currentImageDataURL;
        originalImage.classList.remove('hidden');
        origPlaceholder.classList.add('hidden');
      };
      reader.readAsDataURL(file);
    }

    // ---------- reset results (keep image) ----------
    function resetResults() {
      predictionName.textContent = '—';
      confidenceBadge.textContent = '—';
      confidenceFill.style.width = '0%';
      confidencePercent.textContent = '0%';
      cropName.textContent = '—';
      statusLabel.textContent = '—';
      inferenceTime.textContent = '—';
      top3Items.forEach(item => {
        item.querySelector('span:nth-child(2)').textContent = '—';
        item.querySelector('.bar-fill').style.width = '0%';
        item.querySelector('.prob-text').textContent = '—';
      });
      heatmapImage.classList.add('hidden');
      heatmapPlaceholder.classList.remove('hidden');
      // keep original if exists
      if (currentImageDataURL) {
        originalImage.src = currentImageDataURL;
        originalImage.classList.remove('hidden');
        origPlaceholder.classList.add('hidden');
      } else {
        originalImage.classList.add('hidden');
        origPlaceholder.classList.remove('hidden');
      }
    }

    // ---------- simulate backend diagnosis ----------
    function runDiagnosis() {
      if (isDiagnosing) return;
      if (!currentImageFile || !currentImageDataURL) {
        diagnoseBtn.disabled = true;
        return;
      }

      isDiagnosing = true;
      diagnoseBtn.disabled = true;
      btnText.textContent = 'Analyzing';
      loaderIcon.classList.remove('hidden');

      // pick random disease (simulate prediction)
      const trueDisease = getRandomDisease();
      // generate top3 (including true disease as top1 with high prob)
      const top1 = trueDisease;
      const others = getRandomTop3(top1.name);
      // combine: top1 + 2 others
      let combined = [top1, ...others.slice(0,2)];
      // generate random probs (top1 high)
      let probs = [0.78 + Math.random()*0.18, 0.04 + Math.random()*0.10, 0.02 + Math.random()*0.06];
      const sum = probs.reduce((a,b)=>a+b, 0);
      probs = probs.map(p => p / sum);
      combined = combined.map((d, i) => ({ ...d, prob: probs[i] }));
      combined.sort((a,b) => b.prob - a.prob);

      // simulate inference time
      const inferenceMs = Math.floor(40 + Math.random() * 80);

      setTimeout(() => {
        // update UI
        const top = combined[0];
        predictionName.textContent = top.name;
        const confPct = (top.prob * 100);
        confidenceBadge.textContent = confPct.toFixed(1) + '%';
        confidenceFill.style.width = Math.min(100, confPct) + '%';
        confidencePercent.textContent = confPct.toFixed(1) + '%';
        cropName.textContent = top.crop;
        statusLabel.textContent = top.name.includes('healthy') ? 'Healthy' : 'Disease detected';
        statusLabel.style.color = top.name.includes('healthy') ? '#2a7a3b' : '#b45309';
        inferenceTime.textContent = inferenceMs + ' ms';

        // top3
        top3Items.forEach((item, idx) => {
          const nameSpan = item.querySelector('span:nth-child(2)');
          const barFill = item.querySelector('.bar-fill');
          const probText = item.querySelector('.prob-text');
          if (idx < combined.length) {
            const d = combined[idx];
            nameSpan.textContent = d.name;
            const pct = (d.prob * 100);
            barFill.style.width = Math.max(2, pct) + '%';
            probText.textContent = pct.toFixed(1) + '%';
          } else {
            nameSpan.textContent = '—';
            barFill.style.width = '0%';
            probText.textContent = '—';
          }
        });

        // ----- Grad-CAM heatmap (simulated overlay) -----
        if (currentImageDataURL) {
          const img = new Image();
          img.onload = () => {
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            const w = img.width, h = img.height;
            canvas.width = w; canvas.height = h;
            ctx.drawImage(img, 0, 0, w, h);

            // simulated heatmap overlay
            const gradient = ctx.createRadialGradient(
              w * 0.3, h * 0.35, 10,
              w * 0.5, h * 0.5, w * 0.6
            );
            gradient.addColorStop(0, 'rgba(200, 60, 0, 0.45)');
            gradient.addColorStop(0.4, 'rgba(230, 150, 0, 0.35)');
            gradient.addColorStop(0.8, 'rgba(255, 220, 50, 0.15)');
            gradient.addColorStop(1, 'rgba(255, 255, 200, 0.02)');
            ctx.fillStyle = gradient;
            ctx.fillRect(0, 0, w, h);
            // random spots
            for (let i=0; i<6; i++) {
              const x = w * (0.15 + Math.random()*0.7);
              const y = h * (0.15 + Math.random()*0.7);
              const rad = w * (0.04 + Math.random()*0.10);
              const grd = ctx.createRadialGradient(x, y, 2, x, y, rad);
              grd.addColorStop(0, 'rgba(200, 60, 0, 0.5)');
              grd.addColorStop(1, 'rgba(255, 200, 50, 0)');
              ctx.fillStyle = grd;
              ctx.beginPath();
              ctx.arc(x, y, rad, 0, Math.PI*2);
              ctx.fill();
            }
            const heatmapData = canvas.toDataURL('image/png');
            heatmapImage.src = heatmapData;
            heatmapImage.classList.remove('hidden');
            heatmapPlaceholder.classList.add('hidden');
          };
          img.src = currentImageDataURL;
        } else {
          heatmapImage.classList.add('hidden');
          heatmapPlaceholder.classList.remove('hidden');
        }

        // reset button
        isDiagnosing = false;
        diagnoseBtn.disabled = false;
        btnText.textContent = 'Diagnose';
        loaderIcon.classList.add('hidden');
      }, 900 + Math.random() * 600);
    }

    // ---------- reset everything ----------
    function fullReset() {
      if (isDiagnosing) return;
      fileInput.value = '';
      previewImage.classList.add('hidden');
      placeholderPreview.classList.remove('hidden');
      originalImage.classList.add('hidden');
      origPlaceholder.classList.remove('hidden');

      heatmapPlaceholder.classList.remove('hidden');
      currentImageFile = null;

      diagnoseBtn.disabled = true;

      cropName.textContent = '—';

        item.querySelector('.prob-text').textContent = '—';
      });
    }

    uploadArea.addEventListener('click', () => fileInput.click());

    // init
    resetResults();
    predictionName.textContent = 'Waiting for image';
  })();
